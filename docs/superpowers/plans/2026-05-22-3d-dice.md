## Task 1 — New sound functions

**Context:** The existing `playDiceRoll()` fires a quick rattle + bell when any roll overlay opens. Phase 2 splits this into two distinct sounds: a continuous rolling rattle that lasts the full spin phase (`playDiceRolling`, ~1.15s), and a sharp wooden clack that fires as the first die locks in (`playDiceSettle`, ~0.07s). Both use the existing `tone` and `noise` helpers — no new infrastructure needed.

- [ ] **Step 1: Add `playDiceRolling` to `sounds.ts`**

In `src/lib/sounds.ts`, add after the existing `playDiceRoll` function (~line 153):

```ts
export function playDiceRolling(): void {
  const ctx = getCtx();
  if (!ctx) return;
  const t = ctx.currentTime;
  // 15 noise blips over 1.125s — simulates continuous tumbling
  for (let i = 0; i < 15; i += 1) {
    noise(ctx, t + i * 0.075, 0.06, {
      volume: 0.09,
      bandpass: { freq: 2000 - i * 50, q: 3 },
    });
  }
}
```

- [ ] **Step 2: Add `playDiceSettle` to `sounds.ts`**

Immediately after `playDiceRolling`, add:

```ts
export function playDiceSettle(): void {
  const ctx = getCtx();
  if (!ctx) return;
  const t = ctx.currentTime;
  // Sharp wooden clack: high-freq noise burst + short thud
  noise(ctx, t, 0.04, { volume: 0.3, bandpass: { freq: 2800, q: 1.5 } });
  tone(ctx, 180, t, 0.07, { type: 'sine', volume: 0.35 });
}
```

## Task 2 — Die3D component

**Files:**

- Create: `src/components/ui/Die3D.tsx`

**Context:** `DieFace` is a flat 2D component. `Die3D` replaces it with a true CSS 3D cube: six positioned faces, each rendering their own pip layout. The cube parent uses `transform-style: preserve-3d` and `backface-visibility: hidden` on each face so only the front face is visible. When `variant='spinning'`, the component runs its own `useEffect` interval to cycle random cube rotations — the parent (DiceRollOverlay) continues to cycle `displayDice` values as before, but Die3D ignores the `value` prop during spin and shows the tumbling cube animation instead. When `variant` transitions to `'settled'` or `'highlighted'`, the component cancels its interval and CSS-transitions the cube to the correct face rotation for the given `value`.

**Props interface:** identical to `DieFace` — `value: number`, `variant?: 'spinning' | 'settled' | 'highlighted' | 'wildcard'`, `size?: 'sm' | 'lg'`. Drop-in replacement at all callsites.

**Face layout:** Faces are positioned at the six sides of the cube. Each face renders the pip grid for its own face number (not the `value` prop) — so face 1 always shows 1 pip, face 3 always shows 3 pips, etc. This is correct: when the cube rotates to show face N, you see N pips.

```
Face 1 (front):   translateZ(+half)
Face 6 (back):    rotateY(180deg) translateZ(+half)
Face 2 (right):   rotateY(90deg)  translateZ(+half)
Face 5 (left):    rotateY(-90deg) translateZ(+half)
Face 4 (top):     rotateX(-90deg) translateZ(+half)
Face 3 (bottom):  rotateX(90deg)  translateZ(+half)
```

**Cube rotation to show value N on front:**

```
1 → rotateX(0deg)   rotateY(0deg)
2 → rotateY(-90deg)
3 → rotateX(90deg)
4 → rotateX(-90deg)
5 → rotateY(90deg)
6 → rotateX(180deg)
```

- [ ] **Step 1: Create `src/components/ui/Die3D.tsx`**

```tsx
'use client';

import { useEffect, useRef, useState } from 'react';

const DIE_PIPS: Record<number, [number, number][]> = {
  1: [[1, 1]],
  2: [
    [0, 2],
    [2, 0],
  ],
  3: [
    [0, 2],
    [1, 1],
    [2, 0],
  ],
  4: [
    [0, 0],
    [0, 2],
    [2, 0],
    [2, 2],
  ],
  5: [
    [0, 0],
    [0, 2],
    [1, 1],
    [2, 0],
    [2, 2],
  ],
  6: [
    [0, 0],
    [1, 0],
    [2, 0],
    [0, 2],
    [1, 2],
    [2, 2],
  ],
};

const FACE_ROTATIONS: Record<number, string> = {
  1: 'rotateX(0deg) rotateY(0deg)',
  2: 'rotateY(-90deg)',
  3: 'rotateX(90deg)',
  4: 'rotateX(-90deg)',
  5: 'rotateY(90deg)',
  6: 'rotateX(180deg)',
};

// [face position transform, pip value to render on this face]
const FACES: [string, number][] = [
  ['translateZ(HALF)', 1],
  ['rotateY(180deg) translateZ(HALF)', 6],
  ['rotateY(90deg) translateZ(HALF)', 2],
  ['rotateY(-90deg) translateZ(HALF)', 5],
  ['rotateX(-90deg) translateZ(HALF)', 4],
  ['rotateX(90deg) translateZ(HALF)', 3],
];

function randomRotation(): string {
  const x = Math.round(Math.random() * 360);
  const y = Math.round(Math.random() * 360);
  return `rotateX(${x}deg) rotateY(${y}deg)`;
}

export function Die3D({
  value,
  variant = 'settled',
  size = 'sm',
}: {
  value: number;
  variant?: 'spinning' | 'settled' | 'highlighted' | 'wildcard';
  size?: 'sm' | 'lg';
}) {
  const isWildcard = value === 0 || variant === 'wildcard';
  const isSpinning = variant === 'spinning' && !isWildcard;

  const S = size === 'lg' ? 56 : 28;
  const half = S / 2;
  const borderRadius = size === 'lg' ? '12px' : '6px';
  const pipSize = size === 'lg' ? 'w-3 h-3' : 'w-1.5 h-1.5';
  const gridPad = size === 'lg' ? 'p-2' : 'p-1';

  const [cubeTransform, setCubeTransform] = useState(randomRotation);
  const intervalRef = useRef<ReturnType<typeof setInterval> | null>(null);

  useEffect(() => {
    if (isSpinning) {
      intervalRef.current = setInterval(() => {
        setCubeTransform(randomRotation());
      }, 75);
      return () => {
        if (intervalRef.current) clearInterval(intervalRef.current);
      };
    }
    if (intervalRef.current) {
      clearInterval(intervalRef.current);
      intervalRef.current = null;
    }
    if (!isWildcard) {
      setCubeTransform(FACE_ROTATIONS[value] ?? FACE_ROTATIONS[1]);
    }
  }, [isSpinning, isWildcard, value]);

  const faceClass = isWildcard
    ? 'bg-gray-50 dark:bg-slate-900 border-2 border-dashed border-gray-200 dark:border-slate-700 text-gray-300 dark:text-slate-600'
    : variant === 'spinning'
      ? 'bg-rose-50 dark:bg-rose-950/40 border-2 border-rose-300 dark:border-rose-700 text-rose-500 dark:text-rose-400'
      : variant === 'highlighted'
        ? 'bg-amber-50 dark:bg-amber-950/40 border-2 border-amber-400 dark:border-amber-600 text-amber-600 dark:text-amber-400'
        : 'bg-white dark:bg-slate-900 border-2 border-gray-200 dark:border-slate-700 text-gray-600 dark:text-slate-300';

  const containerClass = [
    'shrink-0 transition-transform duration-150',
    variant === 'highlighted' ? 'scale-110 shadow-md shadow-amber-100 dark:shadow-amber-900' : '',
    variant === 'spinning' ? 'shadow-md shadow-rose-200 dark:shadow-rose-900' : '',
  ]
    .filter(Boolean)
    .join(' ');

  return (
    <div className={containerClass} style={{ width: S, height: S, perspective: S * 5 }}>
      <div
        style={{
          width: S,
          height: S,
          position: 'relative',
          transformStyle: 'preserve-3d',
          transform: cubeTransform,
          transition: isSpinning ? 'none' : 'transform 500ms ease-out',
        }}
      >
        {FACES.map(([faceTx, faceVal]) => (
          <div
            key={faceVal}
            style={{
              position: 'absolute',
              inset: 0,
              transform: faceTx.replace(/HALF/g, `${half}px`),
              backfaceVisibility: 'hidden',
              WebkitBackfaceVisibility: 'hidden',
              borderRadius,
            }}
            className={faceClass}
          >
            {isWildcard ? (
              <div className="flex items-center justify-center w-full h-full text-xs font-bold">
                ?
              </div>
            ) : (
              <div className={`grid grid-cols-3 grid-rows-3 w-full h-full ${gridPad}`}>
                {Array.from({ length: 9 }, (_, idx) => {
                  const row = Math.floor(idx / 3);
                  const col = idx % 3;
                  const pips = DIE_PIPS[faceVal] ?? [];
                  const hasPip = pips.some(([r, c]) => r === row && c === col);
                  return (
                    <div key={idx} className="flex items-center justify-center">
                      {hasPip && <div className={`rounded-full bg-current ${pipSize}`} />}
                    </div>
                  );
                })}
              </div>
            )}
          </div>
        ))}
      </div>
    </div>
  );
}
```
